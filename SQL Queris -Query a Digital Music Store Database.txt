Q1: list of top 5 Artist who have the most number of  Albums?
ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ

SELECT Artist.Name, COUNT(Album.Title) as Total_Album_Num
FROM Artist 
JOIN Album
on Artist.ArtistId=Album.ArtistId
GROUP by Artist.Name
ORDER by 2 DESC
LIMIT 5

Q2: Number of Tracks Sold by Genre?
ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ

SELECT g.name, COUNT(t.trackid) AS Num_Tracks_Sold
FROM genre  g
JOIN Track  t
ON g.genreid = t.genreid
JOIN InvoiceLine  iv
ON iv.trackid = t.trackid
GROUP BY g.name
ORDER BY Nu_Tracks_Sold DESC;

Q3: List Total numbers of Track for each Artist?
ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ

SELECT Artist.Name, count(Track.Name) Total_Tracks_num
FROM Artist
JOIN Album
on Artist.ArtistId=Album.ArtistId
JOIN Track
on Track.AlbumId=Album.AlbumId
GROUP by 1
ORDER by 2 DESC
limit 10;

Q4: Top 10 Albums based on sales?
ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ

SELECT a.albumid, a.title, ar.name AS Artist_Name, SUM(iv.unitprice * iv.quantity) AS Sales
FROM Artist ar
JOIN Album a
ON ar.artistid = a.artistid
JOIN Track t
ON a.albumid = t.albumid
JOIN InvoiceLine iv
ON iv.trackid = t.trackid
GROUP BY a.albumid
ORDER BY Sales DESC
LIMIT 10;
